<html>

  <head>

    <meta charset="UTF-8">

    <title>Matt Holmes Finds the Shooter</title>

    <style>

      .emoji {

        cursor: pointer;

        font-size: 2rem;

        margin: 5px;

      }


      .emoji.selected {

        background-color: red;

      }


      #popup {

        position: absolute;

        top: 50%;

        left: 50%;

        transform: translate(-50%, -50%);

        background-color: #fff;

        padding: 20px;

        border-radius: 10px;

        box-shadow: 0 0 10px rgba(0,0,0,0.5);

      }

    </style>

  </head>

  <body>

    <div id="game-container">

      <p>There is a shooter on the loose, and only Matt Holmes can save you! Once you have run and hid - it’s time to FIGHT. You have three guesses to find the shooter or your colleagues are ALL DEAD!</p>

      <div id="game-board"></div>

      <button id="start-button" onclick="startGame()">Click to Start</button>

      <div id="guesses">Guesses Remaining: 3</div>

      <button id="new-game-button" onclick="newGame()" style="display:none">New Game</button>

    </div>


    <div id="popup" style="display: none;">

      <div id="popup-content"></div>

    </div>


    <script>

      var gameBoard = document.getElementById("game-board");

      var emojis = ["🔫", "💐", "🐶", "🐱", "🐻", "🐼", "🐨", "🐯", "🦁"];

      var shooterIndex = Math.floor(Math.random() * emojis.length);

      var guessesRemaining = 3;


      function startGame() {

        guessesRemaining = 3; // reset guessesRemaining to its original value

        gameBoard.innerHTML = "";

        for (var i = 0; i < emojis.length; i++) {

          gameBoard.innerHTML += "<span class='emoji' onclick='checkGuess(this," + i + ")'>" + emojis[i] + "</span>";

        }

        document.getElementById("start-button").style.display = "none";

      }


      function checkGuess(emojiElement, index) {

        if (guessesRemaining <= 0) {

          return;

        }


        emojiElement.classList.add("selected");


        if (index === shooterIndex) {

          gameBoard.innerHTML = "You found the shooter! All your BeWunder friends are saved!";

          document.getElementById("new-game-button").style.display = "inline-block";

        } else {

          guessesRemaining--;

          document.getElementById("guesses").innerHTML = "Guesses Remaining: " + guessesRemaining;

          if (guessesRemaining === 0) {

            showGameOverPopup("You didn't find the shooter! All your BeWunder friends are dead!", "Try Again");

          }

        }

      }


      function newGame() {

        document.getElementById("new-game-button").style.display = "none";

        startGame();

      }


      function showGameOverPopup(message, buttonLabel) {

        var popup = document.getElementById("popup");

        var popupContent = document.getElementById("popup-content");

        popupContent.innerHTML = message + "<br><button onclick='popup.style.display = \"none\"; startGame();'>" + buttonLabel + "</button>"; // reset guessesRemaining and restart game when button is clicked

        popup.style.display = "block";

      }

    </script>

  </body>

</html>
